127 // Refer to header for documentation
128 void RingSender::update()
129 {
130  if (stat != SENDING)
131  return;
132 
133  for (uint8_t i = 0; i < n_bells; i++) {
134  tx[i].update();
135  }
136 
137  uint8_t _acks = acks();
138  uint8_t _fails = fails();
139  uint8_t total = _acks + _fails;
140 
141  if (total < n_bells)
142  return;
143 
144  if (_fails == n_bells) {
145  log_msg("RingSender::update",
146  "Failed to send ring msg to all " + String(n_bells) + " bells");
147  stat = FAIL;
148  }
149  else if (_acks == n_bells) {
150  log_msg("RingSender::update",
151  "Successfully sent ring msg to all " + String(n_bells) + "/" + String(n_bells) + " bells");
152  stat = SUCCESS;
153  } else {
154  log_msg("RingSender::update",
155  "Failed to send ring msg to " + String(_fails) + "/" + String(n_bells) + " bells");
156  stat = PARTIAL_SUCCESS;
157  }
158 }
