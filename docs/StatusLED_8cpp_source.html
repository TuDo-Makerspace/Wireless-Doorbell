56 void StatusLED::mode(led_mode m) {
57  switch(m) {
58  case OFF:
59  log_msg("StatusLED(PIN:" + String(pin) + ")::mode", "Setting mode to OFF");
60  stat = LOW;
61  break;
62  case ON:
63  log_msg("StatusLED(PIN:" + String(pin) + ")::mode", "Setting mode to ON");
64  stat = HIGH;
65  break;
66  case BLINK: // ON then OFF
67  log_msg("StatusLED(PIN:" + String(pin) + ")::mode", "Setting mode to BLINK");
68  stat = HIGH;
69  tstamp = millis() + blink_interval;
70  break;
71  case BLINK_INV: // OFF then ON
72  log_msg("StatusLED(PIN:" + String(pin) + ")::mode", "Setting mode to BLINK_INV");
73  stat = LOW;
74  tstamp = millis() + blink_interval;
75  m = BLINK;
76  break;
77  default:
78  break;
79  }
80 
81  digitalWrite(pin, stat);
82  mod = m;
83  blks = 0;
84 }

93 void StatusLED::update()
94 {
95  unsigned long t;
96  if (mod == BLINK && (t = millis()) >= tstamp) {
97  stat = !stat;
98  digitalWrite(pin, stat);
99  tstamp = t + blink_interval;
100  if (!stat)
101  blks++;
102  }
103 }
