52  if (!cfg.checkValidity()) {

54  // We can't display error codes if LED pin isn't initialized!

137 Bell::bell_state Bell::connected()
138 {
139  // Calling received() will reset the return value
140  // to false after each call. That way we don't keep
141  // entering this condition when a ring has been received.
142  if (ring_receiver->received()) {
143  led.mode(StatusLED::ON);
144  buzzer.ring();
145  return RINGING;
146  }
147 
148  if (wifi_handler.status() == WiFiHandler::DISCONNECTED)
149  return DISCONNECTED;
150 
151  return CONNECTED;
152 }

166 Bell::bell_state Bell::error()
167 {
168  switch (err) {
169  case UNINITIALIZED: {
170  // update() method called on object that hasn't been
171  // properly initialized/configured. This can occur if
172  // the object is only initialized with the default constructor.
173  // Since no LED pin is initialized, we can't display error codes
174  // and must resort to a fallback error.
175  log_msg("Bell::error", "Bell not initialized!");
176  FALLBACK_ERROR(); // Enters infinite loop
177  break;
178  }
179  case CFG_INVALID: {
180  log_msg("Bell::error", "Invalid configuration provided!");
181  led.setBlinkInterval(250);
182  led.mode(StatusLED::BLINK);
183  break;
184  }
185  }
186 
187  return ERROR_HANDLING;
188 }

201 // Refer to header for documentation
202 void Bell::run()
203 {
204  // The main state machine
205  // Transitions are handled through return values of each state function
206  switch (state) {
207  case INIT: state = init(); break;
208  case DISCONNECTED: state = disconnected(); break;
209  case CONNECTING: state = connecting(); break;
210  case CONNECTED: state = connected(); break;
211  case RINGING: state = ringing(); break;
212  case ERROR: state = error(); break;
213  case ERROR_HANDLING: state = error_handling(); break;
214  }
215 
216  // Update asynchronus components here
217  wifi_handler.update();
218  buzzer.update();
219  led.update();
220 }
