29 /**
30  * @brief RingReceiver class
31  *
32  * The RingReceiver class runs a asynchronous TCP server that listens for incoming
33  * connections, authenticates if the connection is the door transmitter,
34  * and lastly awaits a ring message.
35  *
36  * Once a ring message is received,the received() function will return true.
37  *
38  * To handle incoming connections and data transfers asynchronously, the class
39  * uses callbacks. Since those callbacks are static, the class has been designed
40  * to be a singleton. This ensures all callbacks can access the same instance of
41  * the class.
42  */
43 class RingReceiver {
44 private:
45  inline static AsyncServer *server;
46  inline static AsyncClient *client;
47 
48  inline static IPAddress door_ip;
49  inline static bool running;
50  inline static bool recv;
51 
52  inline static RingReceiver *instance;

54  /**
55  * @brief Default constructor
56  *
57  * The default constructor only serves to allow the class
58  * to be declared without immidiately initializing it. This
59  * spares us from having to allocate the object on the heap via
60  * a pointer, which we generally want to avoid on embedded
61  * systems.
62  */
63  RingReceiver();

67  /**
68  * @brief Callback for new connections
69  *
70  * This callback is called when a new connection is established.
71  * If a client is already connected, or the client is not the door,
72  * the connection is closed and ignored.
73  */
74  static void on_new_client(void* arg, AsyncClient* new_client);
75 
76  /**
77  * @brief Callback for data received from client
78  *
79  * This callback is called when data is received from the client.
80  * If the data is a ring message, the received() function will return true.
81  * If the data is invalid, the connection is closed and ignored.
82  */
83  static void on_data(void* arg, AsyncClient* client, void *data, size_t len);

92  static void on_disconnect(void* arg, AsyncClient* client);

117  static RingReceiver *get_instance();

119  /**

122  * This function starts the ring receiver. It will listen for incoming

125  *
126  * @param port The port to listen on
127  * @param door_ip The IP address of the door transmitter
128  */
129  void begin(uint16_t port, String door_ip_addr);
130 
131  /**
132  * @brief Returns if a ring message has been received
133  *
134  * This function returns true if a ring message has been received.
135  * Calling the function will reset the return value to false.
136  * Once called, the function will reset the received value to false,

139  * Poll this function to check if a ring message has been received.
140  *
141  * @returns true if a new ring message has been received,
142  * false if no new ring message has been received
143  */
