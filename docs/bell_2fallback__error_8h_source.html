32 /**
33  * @brief Bell Fallback error
34  *
35  * The following function is called when the Bell class
36  * encounters an error and has no possibility to display
37  * it to the user via the LED (eg. no LED pin provided in
38  * the class configuration).
39  *
40  * The fallback error has direct access to the LED and
41  * buzzer pins, and will blink the LED and play a A4
42  * tone on the buzzer indefinitely.
43  *
44  * @attention This function runs in an infinite loop!
45 */
46 inline void FALLBACK_ERROR()
47 {
48  log_msg("FALLBACK_ERROR", "Entered fallback error mode, something went horribly wrong!");
49 
50  pinMode(BELL_LED, OUTPUT);
51  pinMode(BELL_BUZZER, OUTPUT);
52 
53  while(true) {
54 #ifndef BELL_SILENT
55  tone(BELL_BUZZER, NOTE_A4);
56 #endif
57  digitalWrite(BELL_LED, HIGH);
58  delay(250);
59  noTone(BELL_BUZZER);
60  digitalWrite(BELL_LED, LOW);
61  delay(250);
62  }
63 }
