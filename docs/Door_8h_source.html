36 /**
37  * @brief The Main Door class
38  *
39  * The Door class handles the main logic of the Door transmitter.
40  * This includes initializing and handling all sub classes (RingSender,
41  * WiFiHandler, Status LEDs etc.), printing the boot message, handling errors

85  /**
86  * @brief Entry point of the state machine
87  *
88  * The init function is the entry point of the state machine.
89  * It enables the power LED, prints the boot message, and tells
90  * the WiFiHandler to connect to the WiFi network.
91  *
92  * Once complete, the init method returns the CONNECTING state,
93  * which the state machine will then enter.
94  *
95  * @returns CONNECTING, the next state of the state machine.
96  */
97  door_state init();
98 
99  /**
100  * @brief Handles the CONNECTING state
101  *
102  * The connecting() function handles the CONNECTING state.
103  * The CONNECTING state will poll the WiFiHandler to check
104  * if the connection to the WiFi network has been established.
105  *
106  * If the connection has been established, the connecting()
107  * function will return the CONNECTED state.
108  *
109  * If the connection attempt has timed out, the connecting()
110  * function will return the ERROR state and set the error type
111  * to NO_WIFI.
112  *
113  * @returns The next state of the state machine:
114  * CONNECTING, if the WiFiHandler is still connecting.
115  * CONNECTED, if the WiFiHandler has connected.
116  * ERROR (NO_WIFI), if the WiFiHandler has timed out.
117  */
118  door_state connecting();
119 
120  /**
121  * @brief Handles the CONNECTED state
122  *
123  * The connected() function handles the CONNECTED state.
124  * The CONNECTED state sends a ring message to all bells
125  * via the RingSender class and then returns the RINGING
126  * state.
127  *
128  * @returns RIGHTING, the next state of the state machine.
129  */
130  door_state connected();
131 
132  /**
133  * @brief Handles the RINGING state
134  *
135  * If the RingSender is still awaiting a response, the RIGHTING
136  * state will be returned.
137  *
138  * If all bells have been rang, the connected() function will
139  * return the POWER_OFF state.
140  *
141  * If some bells have been rang, but not all, the connected()
142  * function will return the ERROR state and set the error type
143  * to PARTIAL_SUCCESS.
144  *
145  * If no bells have been rang, the connected() function will
146  * return the ERROR state and set the error type to FAIL.
147  *
148  * @returns The next state of the state machine:
149  * RINGING, if the RingSender is still awaiting a response.
150  * POWER_OFF, if all bells have been rang.
151  * ERROR (PARTIAL_SUCCESS), if some bells have been rang, but not all.
152  * ERROR (FAIL), if no bells have been rang.
153  */
154  door_state ringing();
155 
156  /**
157  * @brief Handles the ERROR state
158  *
159  * The error() function handles the ERROR state.
160  * If an error occurs, the state machine transitions to the
161  * ERROR state. The ERROR state determines the type of error
162  * and prepares the error handling (eg. setting status LEDs to blink mode).
163  *
164  * The error is then handled in the error_handling() function.
165  *
166  * @returns ERROR_HANDLING, the next state of the state machine.
167  */
168  door_state error();
169 
170  /**
171  * @brief Handles the ERROR_HANDLING state
172  *
173  * The error_handling() function handles the ERROR_HANDLING state.
174  * The ERROR_HANDLING state handles the error by blinking the LED
175  * and waiting for the user to reset the device.
176  *
177  * @returns ERROR_HANDLING, if the error is still being handled or is fatal.
178  * POWER_OFF, once the error has been handled.
179  */
180  door_state error_handling();
181 
182  /**
183  * @brief Handles the POWER_OFF state
184  *
185  * The power_off() function handles the POWER_OFF state.
186  * The POWER_OFF state unlatches the power latch and returns
187  * the POWERED_OFF state.
188  *
189  * Ideally the POWER_OFF state should never be reached, as the
190  * device should be powered off by then. However, if a devboard
191  * is used, or the power isn't immediately cut off, the POWER_OFF
192  * state will be reached, where the device reaches an infinite
193  * loop.
194  *
195  * @returns POWERED_OFF, the next state of the state machine.
196  */
197  door_state power_off();
