29 /**
30  * @brief StatusLED class
31  *
32  * The StatusLED class provides a simple "asynchronous" interface for controlling status LEDs.
33  * LEDs can be set to blink, be on or off. In order to work "asynchronously", the class
34  * requires the user to continously the update() method.
35  */
36 class StatusLED {
37 public:
38  /// Possible LED states
39  enum led_mode {
40  OFF, ///< Constantly off
41  ON, ///< Constantly on
42  BLINK, ///< First ON then OFF
43  BLINK_INV ///< First OFF then ON
44  };

57  * The default constructor only serves to allow the class
58  * to be declared without immidiately initializing it. This
59  * spares us from having to allocate the object on the heap via
60  * a pointer, which we generally want to avoid on embedded
61  * systems.
62  */
63  StatusLED();
64 
65  /**
66  * @brief Constructor
67  *
68  * The constructor initializes the StatusLED object with the
69  * given pin and blink interval.
70  *
71  * The blink interval defines the time between two consecutive
72  * LED state changes if the LED is set to blink.
73  *
74  * @param pin The pin to which the LED is connected
75  * @param blink_interval The blink interval in milliseconds (default: 500)
