78 RingTX::ring_stat RingTX::con()
79 {
80  if (client.connected()) {
81  log_msg("RingTX(to:" + ip + ":" + String(port) + ")::con",
82  "Connected to bell at " + ip + ":" + String(port));
83  tstamp = millis() + timeout;
84  return SENDING;
85  }
86 
87  if (timeout && millis() >= tstamp) {
88  log_msg("RingTX(to:" + ip + ":" + String(port) + ")::con",
89  "Failed to connect to bell at " + ip + ":" + String(port) + ", timed out!");
90  return FAIL;
91  }
92 
93  return CONNECTING;
94 }

97 RingTX::ring_stat RingTX::sen()
98 {
99  if (txRingMSG())

102  if (timeout && millis() >= tstamp) {
103  log_msg("RingTX(to:" + ip + ":" + String(port) + ")::send",
104  "Failed to send ring msg to bell at " + ip + ":" + String(port) + ", timed out!");
105  return FAIL;
