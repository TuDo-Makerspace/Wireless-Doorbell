49 Door::Door(DoorCFG door_cfg) : cfg(door_cfg)
50 {
51  log_msg("Door::Door", "Initializing door");
52 
53  if (!cfg.checkValidity()) {
54  if (cfg.ring_led_pin == -1 ||
55  cfg.power_led_pin == -1) {
56  log_msg("Door", "Status LEDs uninitialized, entering fallback error mode");

59 
60  err = CFG_INVALID;

62  return;

65  ring_led = StatusLED(cfg.ring_led_pin);
66  pwr_led = StatusLED(cfg.power_led_pin);
67 
68  IPAddress ip, gateway, subnet;
69  ip.fromString(cfg.static_ip);
70  gateway.fromString(cfg.gateway);
71  subnet.fromString(cfg.subnet);
72 
73  wifi_handler = WiFiHandler(
74  cfg.ssid, cfg.psk,
75  ip, gateway, subnet,
76  cfg.con_timeout_s, false
77  );
78 
79  ring_sender = RingSender(ip, cfg.port, cfg.n_bells, cfg.bell_timeout_ms);

81  state = INIT;
82 }

137 Door::door_state Door::ringing()
138 {
139  switch (ring_sender.status()) {
140  case RingSender::SENDING: {
141  if (ring_sender.acks() > 0 &&
142  ring_led.getMode() != StatusLED::ON) {
143  ring_led.mode(StatusLED::ON);
144  }
145  return RINGING;
146  }
147 
148  case RingSender::SUCCESS: {
149  ring_led.mode(StatusLED::OFF);
150  return POWER_OFF;
151  }
152 
153  case RingSender::PARTIAL_SUCCESS: {
154  ring_led.mode(StatusLED::OFF);
155  err = PARTIAL_SUCCESS;
156  return ERROR;
157  }
158 
159  default: {
160  err = FAIL;
161  return ERROR;
162  }
163  }
164 }

167 Door::door_state Door::error()
168 {
169  switch (err) {
170  case UNINITIALIZED:
171  log_msg("Door::error", "Door not initialized!");
172  FALLBACK_ERROR();
173  break;
174  case CFG_INVALID:
175  log_msg("Door::error", "Invalid configuration provided!");
176  pwr_led.setBlinkInterval(250);
177  pwr_led.mode(StatusLED::BLINK);
178  break;
179  case NO_WIFI:
180  log_msg("Door::error", "Failed to establish a WiFi connection!");
181  pwr_led.mode(StatusLED::BLINK_INV);
182  ring_led.mode(StatusLED::BLINK);
183  break;
184  case PARTIAL_SUCCESS:
185  log_msg("Door::error", "Partial success, some bells did not ring!");
186  pwr_led.mode(StatusLED::BLINK);
187  ring_led.mode(StatusLED::BLINK);
188  break;
189  case FAIL:
190  log_msg("Door::error", "Failed to contact bells!");
191  pwr_led.mode(StatusLED::BLINK);
192  break;
193  }
194 
195  return ERROR_HANDLING;
196 }

199 Door::door_state Door::error_handling()
200 {
201  switch(err) {
202  case CFG_INVALID: {
203  if (pwr_led.blinks() >= DOOR_CFG_INVALID_BLINKS) {
204  return POWER_OFF;
205  }
206  break;
207  }
208 
209  case NO_WIFI: {
210  if (pwr_led.blinks() >= DOOR_NO_WIFI_BLINKS &&
211  ring_led.blinks() >= DOOR_NO_WIFI_BLINKS) {
212  return POWER_OFF;
213  }
214  break;
215  }
216 
217  case PARTIAL_SUCCESS: {
218  if (pwr_led.blinks() >= DOOR_PARTIAL_SUCCESS_BLINKS &&
219  ring_led.blinks() >= DOOR_PARTIAL_SUCCESS_BLINKS) {
220  return POWER_OFF;
221  }
222  break;
223  }
224 
225  case FAIL: {
226  if (pwr_led.blinks() >= DOOR_NO_BELLS_BLINKS) {
227  return POWER_OFF;
228  }
229  break;
230  }
231 
232  default: {
233  break;
234  }
235  }
236 
237  return ERROR_HANDLING;
238 }

253 // Refer to header for documentation
254 void Door::run()
255 {
256  switch (state) {
257  case INIT: state = init(); break;
258  case CONNECTING: state = connecting(); break;
259  case CONNECTED: state = connected(); break;
260  case RINGING: state = ringing(); break;
261  case ERROR: state = error(); break;
262  case ERROR_HANDLING: state = error_handling(); break;
263  case POWER_OFF: state = power_off(); break;
264  case POWERED_OFF: break;
265  }
266 
267  wifi_handler.update();
268  ring_sender.update();
269  pwr_led.update();
270  ring_led.update();
271 }
