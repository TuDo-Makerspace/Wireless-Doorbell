75 void RingReceiver::on_new_client(void* arg, AsyncClient* new_client)
76 {
77  IPAddress ip = new_client->remoteIP();
78 
79  log_msg("handleNewClient", "New client connected with IP: " + ip.toString());
80 
81  // Since the door is the only client we expect, we can just ignore any other clients
82  // Theoretically, we should not be receiving any other clients than the door, but
83  // just in case some goofball tries to connect to the bell, we'll just ignore them
84 
85  if (client != NULL) {
86  log_msg("RingReceiver::on_new_client", "Client already connected! Ignoring new client...");
87  new_client->close();
88  return;
89  }
90 
91  if (ip != door_ip) {
92  log_msg("RingReceiver::on_new_client", "Client is not the door! Ignoring new client...");
93  new_client->close();
94  return;
95  }
96 
97  client = new_client;
98 
99  log_msg("RingReceiver::on_new_client", "Client is the door!");
100 
101  // Register callbacks for client events
102  client->onData(&on_data, NULL);
103  client->onDisconnect(&on_disconnect, NULL);
104  client->onTimeout(&on_timeout, NULL);
105  client->onError(&on_error, NULL);
106 }

109 void RingReceiver::on_data(void* arg, AsyncClient* client, void *data, size_t len)
110 {
111  log_msg("RingReceiver::on_data", "Received data from door");
112 
113  uint8_t msg;
114 
115  if (len != 1)
116  goto INVALID_PACKET;
117 
118  // Cast data to uint8_t as the ring_msg is only 1 byte
119  msg = *((uint8_t *)data);
120 
121  if (msg == RING_MSG) {
122  recv = true;
123  log_msg("RingReceiver::on_data", "Received ring message from door");
124  } else {
125  goto INVALID_PACKET;
126  }
127 
128  log_msg("RingReceiver::on_data", "Closing connection with door");
129  client->close();
130 
131  return;
132 
133  INVALID_PACKET:
134  log_msg("RingReceiver::on_data", "Invalid packet received from door! Closing connection!");
135  client->close();
136 }
